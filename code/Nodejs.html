# Node.js

​ NodeJS是开发服务器后台的东西，与PHP、JavaEE、python类似，和传统的浏览器的关注DOM的JS完全不同，将JavaScript触角伸到了服务器端。内核是Chrome浏览器的V8引擎，解析JavaScript的效率是非常快的

Node是对js标准的一个实现,Node也是一个js引擎

通过Node可以使js代码在服务器端执行

Node仅仅对ES标准进行实现,所在Node中不包含DOM 和 BOM

Node中可以使用所有的内建对象

​ String Number Boolean Math Date RegExp Function Object Array

​ 而BOM 和DOM都不能使用

​ 但是可以使用console 也可以使用定时器(setTimeout( ) setInterval( ))

Node可以在后台来编写服务器

​ Node编写服务器都是单线程的服务器

Node的服务器单线程的

​ Node处理请求时是单线程的,但是后台拥有一个I/O线程池

//参数返回是一个布尔值,为真则是有,反之则是没有

    console.log(flag? '有':'没有');

# 模块规范(自定义模块)

​ 在一个项目下可能会用到很多个的包,可以自定义包,每个自定义的文件都相当于一个独立的小房子,想要使用里面的工具需要这个文件开门,进行暴露。暴露后的方法能够被外部访问，没有暴露的绝对不能被访问。在入口文件使用就能访问到这个文件的方法，每个文件都能相互访问。在每个文件的执行中，发现访问其他文件的代码时，会先解析其他文件的属性和方法，解析过后才会继续解析自己的代码

暴露方法

module.exports
//在暴露功能过多时可以暴露出一个对象
module.exports={
    方法1:add,
    方法2:reduce
}
方法2

# post请求

需要用到的包 : express 设置服务器

​ body-parser 快速设置请求头

express没有处理post请求参数的工具 需要引入body-parser中间件

var express=require("express");
var bodyParser=require("body-parser");
var app=express();
// 处理请求体中携带的参数    req.body

//设置中间件
// parse application/x-www-form-urlencoded
app.use(bodyParser.urlencoded({ extended: false }))
// parse application/json (切割方式)
app.use(bodyParser.json())


/*请求区别*/
app.get("/goodList",function(req,res){
    res.json(req.query);
})


// post express没有处理post请求参数  需要引入body-parser中间件 req.body
app.post("/login",function(req,res){
    //打印请求头信息(所有)
    console.log(req.headers);
    //返回信息
    res.json(req.body)
})
//开启
app.listen(8080);

# 代理服务器

需要用到的包 : express, http-proxy-middleware

express构建服务器。

http-proxy-middleware 使用http-proxy-middleware下面的 createProxyMiddleware 方法设置代理。

var express=require("express");
var app=express();
// 引入代理模块
var proxy=require('http-proxy-middleware').createProxyMiddleware;
//设置中间件
var proxyFun=proxy({
    // 目标服务器地址
    target:"https://www.kuwo.cn",
    changeOrigin: true,
    // 路径改写  处理标识(在实际请求中把/zhangsan替换成"")
    pathRewrite:{
        "^/zhangsan":""
    },
    // 每次代理之前  设置请求需要的请求头
    onProxyReq:function(proxyReq){
        proxyReq.setHeader("csrf","CAQBBHLL57M");
        proxyReq.setHeader("Cookie","kw_token=CAQBBHLL57M")
    }
})
// 当请求地址以/zhangsan开头时 使用此中间件
//设置代理服务器的正常转换情况
/*
    真实请求地址
        http://www.a.com/goodList
    
        target: http://www.a.com

        /a 开头的需要代理
        前端： /a/goodList
        代理：target+/a/goodList = http://www.a.com/a/goodList
*/

//当请求地址以/zhangsan开头时,使用proxyFun中间件
app.use("/zhangsan",proxyFun)

// 指定静态资源目录
app.use(express.static("public"))
//不使用代理服务器情况下的正常访问
app.get("/abc",function(req,res){
    res.json({msg:"abc"})
})
//设置端口
app.listen(6060);

# mysql

sql语句

🥇增

  insert into 表名   (字段名1,字段名2) values(值1,值2)
  insert into users (userName,password) values("张三","123456")
  insert into users (userName,password) values("李四","abcd")

🥈删

  delete from 表名  where 字段=值
  delete from users where userName="张三"

🥉改

  update 表名 set 字段=新值 where userName="李四"
  update users set 字段1=值1,字段2=值2 where userName="李四"
  update users set password="abcd",userName="张三" where userName="李四"

4️⃣查

 /*		所有 *    表单 */
  select * from users  （所有字段）
  /*查看users表单下所有的userName*/
  select userName from users  (某个字段)
  select userName,password from users  (多个字段)
  /*查找表单里面userName是张三的*/
  select * from users where userName="张三" （指定范围）

5️⃣ where语句

where子语句 条件 = > < != and or

and 且

or 或

/*查找 users 表单中所有 Id>2 且 Id<4*/
select * feom users where Id>2 and Id<4

6️⃣分页

limit 0,3 从第二条开始向后查3条 不包含第二条

/*查找goods中所有的数据展示0到50*/
select * from goods limit 0,50

7️⃣排序

order by 字段名(按什么字段排序) desc (desc降序 asc升序)

/*查找goods中所有的数据以 price字段进行升序排列*/
select * form goods order by price asc

8️⃣in

in 多选型

​ 在一个条件具多个选择时

/*查找book表单中所有的userId=1或userId=2的*/
select * from book where userId=1 or userId=2
in写法
select * from book where userId in(1,2)

表达式也可以当成条件

select * from users where password=(select password from users where userName="王二") and userName!="王二"
/*in*/
select * from book where userId in (select Id from users where username in ("张三","李四"))
