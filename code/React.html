事件传参

**难点： ** bind修改this指向写法event对象会在最后一个参数

<script type="text/babel">
  const root = ReactDOM.createRoot(document.querySelector('#root'))
  class App extends React.Component {
    // class fields
    name = 'App'
    constructor() {
      super()
      this.state = {
        message: 'React',
        count: 0,
        name: 'kobe'
      }
    }
    foo(e) {
      console.log('event', e,this);
      this.setState({
        message: 'wenn'
      })
    }
    fun(e, name, age) {
      console.log('event', e,name, age);
    }
    render() {
      const { message, count, name } = this.state
      return (
        <div>
          {/* event 参数传递 */}
          <button onClick={this.foo.bind(this)} >event参数传递</button>
          {/* 箭头函数event参数传递 */}
          <button onClick={(e) => this.foo(e)} >参数传递</button>
          {/* 额外的参数传递 */}
          <button onClick={(e) => this.fun(e, 'wenn', 22)} >额外参数传递</button>
          {/* bind传参event对象会在最后 */}
          <button onClick={this.fun.bind(this, 'wenn', 22)} >额外参数传递(不推荐)</button>
        </div>
      )
    }
  }
  root.render(<App />)
</script>

# PWA

https://developer.mozilla.org/zh-CN/docs/Web/Progressive_web_apps

  1. PWA全称称Progressive Web App，即渐进式WEB应用
  2. 一个 PWA 应用首先是一个网页, 可以通过 Web 技术编写出一个网页应用
  3. 随后添加上 App Manifest 和 Service Worker 来实现 PWA 的安装和离线等功能
  4. 这种Web存在的形式，我们也称之为是 Web App

PWA解决了哪些问题

  1. 可以添加图标至主屏幕，点击主屏幕图标可以实现启动动画以及隐藏地址栏
  2. 实现离线缓存功能，即使用户手机没有网络，依然可以使用一些离线功能
  3. 实现了消息推送
  4. 实现一系列类似于Native App相关的功能

# 组件

React组件化（相比于Vue更加灵活）

  1. 根据组件的定义方式，可以分为：函数组件(Functional Component )和类组件(Class Component)
  2. 根据组件内部是否有状态需要维护，可以分成：无状态组件(Stateless Component )和有状态组件(Stateful Component)
  3. 根据组件的不同职责，可以分成：展示型组件(Presentational Component)和容器型组件(Container Component)

函数组件、无状态组件、展示型组件主要关注UI的展示

类组件、有状态组件、容器型组件主要关注数据逻辑

# 非父子组件通信（Context）

  1. 首先要创建一个Context
import React from 'react'
const ThemeContext = React.createContext()
export default ThemeContext
  1. 在需要的组件中引入创建的Context
    1. 通过ThemeContext中Provider 中的value 属性为后代提供数据 （需要使用数据的组件被标签包裹）
    2. 可以设置多个Context同样使用标签包裹
import React, { Component } from 'react'
import Home from './Home'
import ThemeContext from './context/theme-context'
import userContext from './context/user-context'
class App extends Component {
  constructor() {
    super()
    this.state = {
      info: { name: 'coder', age: 22 }
    }
  }
  render() {
    const { info } = this.state
    return (
      <div>
        <userContext.Provider value={{name: 'kobe'}}>
          <ThemeContext.Provider value={{ color: 'red', size: 30 }}>
            <Home {...info}></Home>
          </ThemeContext.Provider>
        </userContext.Provider>
      </div>
    )
  }
}
export default App
  1. 在子组件中同样引入Context 设置contextType 为我们某一个Context
    1. 假如有多个Context但是contextType只能设置一个这时可以使用Consumer 花括号中的函数value参数就是父组件传入的值
    2. contextType 接收的值可以用this.context

# 生命周期

记录点:

getSnapshotBeforeUpdate返回值会在componentDidUpdate的第三个参数访问到

shouldComponentUpdate的返回值决定数据修改时页面会不会刷新（render会不会执行）

// 声明周期
import { Component } from "react";
class Cycle extends Component {
  // 先执行constructor
  constructor() {
    super()
    this.state = {
      message: 'React'
    }
  }
  componentDidMount() {
    // 发送网络请求比较合适
    // 添加一些订阅时
    // 组件挂载到DOM上时
    console.log('组件挂载DOM时: DidMount');
  }
  componentDidUpdate(prevProps, prevState, snapshot) {
    /*
      prevState: 前一个state
      snapshot: getSnapshotBeforeUpdate 返回的数据
    */
    console.log(prevProps, prevState, snapshot)
    // 组件数据修改时
    console.log('组件修改时: DidUpdate');
  }
  componentWillUnmount() {
    // 组件卸载时
    console.log('组件被卸载: WillUnmount');
  }
  // 不常用生命周期
  shouldComponentUpdate() {
    // 决定 render函数是否从新渲染
    // 返回 false 不从新渲染  true 重新渲染
    return true
  }
  getSnapshotBeforeUpdate() {
    // 在这里返回的东西可以在 componentDidUpdate 拿到
    return {
      useScroll: 1000
    }
  }
  modify() {
    this.setState({
      message: 'Vue'
    })
  }
  render() {
    const { message } = this.state
    return (
      <div>
        声明周期
        {message}
        <button onClick={() => { this.modify() }}>修改</button>
      </div>
    )
  }

}

export default Cycle
