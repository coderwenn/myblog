never

表示参数没有这个类型

与void的区别: void表示没有任何类型,never表示永远不存在值的类型.当一个函数返回空值时，它的返回值为 void 类型，但是，当一个函数永不返回时（或者总是抛出错误），它的返回值为 never 类型。void 类型可以被赋值，但是除了 never 本身以外，其他任何类型不能赋值给 never。

function fun():never {

}

  throw new Error('Throw my hands in the air like I just dont care');

function fail(message: string): never {
  throw new Error(message);
}

class Student extends Person {
  sno: number

    super(name,age)
    this.sno = sno
  }
  studying() {
    console.log('学习')
  }
  // 父类方法的重写
  eating() {
    // 同时执行父类的方法
    super.eating()
    console.log('Studenteating');
  }
}

stu.eating()

abstract class Shape {
  abstract getArea()
}

class Rectangle extends Shape {
  private w: number
  private h: number
  constructor(w: number, h: number) {
    super()
    this.w = w
    this.h = h
  }
  // 抽象类的方法必须被子类实现 除非之类也是一个抽象类

  }
}

// 类型的参数化
// 定义函数时,不决定这些参数的类型
// 让调用者以参数的形式告知，这个函数时什么类型
function sum<T>(num1: T): T {
  return num1
}
// 调用方式1: 明确的传入类型
sum<number>(1)
sum<any[]>(['wqx'])
sum<{name: string}>({name: 'wqx'})

// 调用方式2: 类型推导

sum(50)
sum([123])

泛型接受参数类型

function foo<T, E>(arg1: T, arg2: E) {
  return arg1
}
foo<number,string>(10,'20')

泛型接口

interface IPerson<T1, T2> {
  name: T1
  age: T2
}

const p: IPerson<string, number> = {
  name: 'wqx',
  age: 22
}

泛型类的使用

class Point<T> {
  x: T
  y: T
  z?: T
  constructor(x: T, y: T, z?: T) {
    this.x = x
    this.y = y
    this.z = y
  }
}

/* 调用方法:
    隐私推导
    显示
*/
const p1 = new Point('1.33', '1.33', '1.33')
const p2 = new Point<string>('1.33', '1.33', '1.33')
const p3: Point<string> = new Point('1.33', '1.33', '1.33')
